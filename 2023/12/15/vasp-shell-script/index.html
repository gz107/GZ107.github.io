[2] 批量化的任务提交

vi shell

#复制及提交任务，假设现在有一个egl文件#
#! /bin/bash

for i in 300 350 400 450 500
do
cp -r egl $i #复制egl文件命名为i(1 2 3 4)
cd $i #进入该文件
echo “ENCUT=$i” >> INCAR #>>表示输出重定向
echo “ISIF=3” >>INCAR
sbatch JOB
cd ..

done

:wq #保存输入的文件
